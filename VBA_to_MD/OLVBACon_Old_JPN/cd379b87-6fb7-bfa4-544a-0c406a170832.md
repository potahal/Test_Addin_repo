Private Sub myFolders_FolderChange(ByVal Folder As Outlook.Folder) 
 
 If Folder.Items.Count = 0 Then 
 
 MyPrompt = Folder.Name &amp; " is empty. Do you want to delete it?" 
 
 If MsgBox(MyPrompt, vbYesNo + vbQuestion) = vbYes Then 
 
 Folder.Delete 
 
 End If 
 
 End If 
 
End Sub
